public string GetGroupingHeader(Barcode barcode)
{
    var date = barcode.Date.Date;

    if (date == today)
        return "Today";

    if (date == yesterday)
        return "Yesterday";

    if (lastWeek <= date && date < yesterday)
        return "Last week";

    if (date.Year == today.Year)
        return date.ToString("MMMM");

    if (date.Year == today.Year - 1)
        return "Last year";

    return $"{date.Year}";
}
